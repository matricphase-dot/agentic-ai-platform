    <script>
        // Load dashboard data
        document.addEventListener('DOMContentLoaded', function() {
            loadDashboardData();
            loadRecentActivity();
            
            // Auto-refresh every 30 seconds
            setInterval(loadDashboardData, 30000);
        });
        
        async function loadDashboardData() {
            try {
                // Load system stats
                const response = await fetch('/api/system-stats');
                const data = await response.json();
                
                document.getElementById('aiRequests').textContent = data.ai_requests || '0';
                document.getElementById('filesOrganized').textContent = data.files_organized || '0';
                document.getElementById('timeSaved').textContent = data.time_saved_minutes || '0';
                document.getElementById('recordings').textContent = data.recordings || '0';
                
            } catch (error) {
                console.error('Error loading dashboard data:', error);
            }
        }
        
        async function loadRecentActivity() {
            try {
                const response = await fetch('/api/analytics/recent');
                const data = await response.json();
                
                const activityList = document.getElementById('activityList');
                
                if (data.activities && data.activities.length > 0) {
                    let html = '';
                    data.activities.forEach(activity => {
                        html += `
                            <div class="activity-item mb-3 p-3 border rounded">
                                <div class="d-flex justify-content-between align-items-center">
                                    <div>
                                        <i class="fas fa-${activity.icon} me-2"></i>
                                        <strong>${activity.title}</strong>
                                        <p class="mb-0 text-muted">${activity.description}</p>
                                    </div>
                                    <small class="text-muted">${activity.time}</small>
                                </div>
                            </div>
                        `;
                    });
                    activityList.innerHTML = html;
                } else {
                    activityList.innerHTML = `
                        <div class="text-center py-4">
                            <i class="fas fa-history fa-2x mb-3 text-muted"></i>
                            <p>No recent activity yet. Start using the platform!</p>
                        </div>
                    `;
                }
                
            } catch (error) {
                console.error('Error loading activity:', error);
            }
        }
        
        function quickStart() {
            // Start a quick automation workflow
            alert('Starting quick automation workflow... Check the console for progress.');
            console.log('Quick start initiated');
        }
        
        async function startRecording() {
            try {
                const response = await fetch('/api/desktop/start-recording', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        quality: 'medium',
                        fps: 30,
                        audio: 'system'
                    })
                });
                
                const data = await response.json();
                alert(data.message || 'Recording started!');
            } catch (error) {
                console.error('Error starting recording:', error);
            }
        }
        
        function organizeFiles() {
            window.location.href = '/file-organizer';
        }
        
        function openAI() {
            window.location.href = '/ai-automation';
        }
        
        async function generateQR() {
            try {
                const response = await fetch('/api/mobile/qr');
                const data = await response.json();
                
                if (data.qr_url) {
                    // Show QR code in modal
                    const modal = `
                        <div class="modal fade" id="qrModal" tabindex="-1">
                            <div class="modal-dialog">
                                <div class="modal-content">
                                    <div class="modal-header">
                                        <h5 class="modal-title">Scan QR to Pair Mobile</h5>
                                        <button type="button" class="btn-close" data-bs-dismiss="modal"></button>
                                    </div>
                                    <div class="modal-body text-center">
                                        <img src="${data.qr_url}" alt="QR Code" class="img-fluid mb-3">
                                        <p>Scan with your mobile device to pair</p>
                                        <p class="text-muted small">Pairing Code: ${data.pairing_code || 'N/A'}</p>
                                    </div>
                                </div>
                            </div>
                        </div>
                    `;
                    
                    document.body.insertAdjacentHTML('beforeend', modal);
                    new bootstrap.Modal(document.getElementById('qrModal')).show();
                }
            } catch (error) {
                console.error('Error generating QR:', error);
            }
        }
        
        // Add Bootstrap if not already loaded
        if (!window.bootstrap) {
            const script = document.createElement('script');
            script.src = 'https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/js/bootstrap.bundle.min.js';
            document.head.appendChild(script);
        }
    </script>
